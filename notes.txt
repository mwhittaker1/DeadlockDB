def create_player_rolling_stats(con):
    con.execute("""
    CREATE TABLE player_rolling_stats (
    account_id BIGINT,
    match_id BIGINT,
    start_time TIMESTAMP,
    p_win_pct_2 FLOAT,
    p_win_pct_3 FLOAT,
    p_win_pct_4 FLOAT,
    p_win_pct_5 FLOAT,
    p_loss_pct_2 FLOAT,
    p_loss_pct_3 FLOAT,
    p_loss_pct_4 FLOAT,
    p_loss_pct_5 FLOAT,
    PRIMARY KEY (account_id, match_id),
    FOREIGN KEY (account_id, match_id) REFERENCES player_matches(account_id, match_id)
